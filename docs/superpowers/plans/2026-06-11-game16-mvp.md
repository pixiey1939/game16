# 解谜游戏16：消失的麻姐 — MVP 实现计划

> **面向 AI 代理的工作者：** 必需子技能：使用 superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans 逐任务实现此计划。步骤使用复选框（`- [ ]`）语法来跟踪进度。

**目标：** 从 `script/game_design.txt`（v7.0）和 `script/第六部分：游戏流程.md`（v2.0）设计文档实现一个能跑通 5 阶段主流程的 PWA 可玩原型，21 证据、4 Combine、4 结局全部可触发。

**架构：** 单状态机 + 命令分发。所有证据数据内嵌在 `js/evidence.js` 为单一 JS 对象，玩家通过 `access/list/combine/conclusions/backup/submit/clear` 7 个命令交互（加 `help` 共 8 个）。GameState 由 `js/game.js` 集中管理。

**技术栈：** 纯 HTML5 + CSS3 + 原生 JavaScript（ES2020+）。无框架、无外部依赖。localStorage 持久化。多文件组织（GitHub Pages 友好）。**不**做 PWA/Service Worker。

**参考规格：** `docs/superpowers/specs/2026-06-11-game16-mvp-design.md`

```javascript
// GAME 16 — UI 工具集
// 提供 print / printHtml / animate / startScreen / cursorBlink 等无状态工具函数。
// 依赖：DOM 元素（#output）和全局 GameState

const ui = (() => {
  // 缓存 DOM 引用
  let outputEl = null;

  function ensureOutput() {
    if (!outputEl) outputEl = document.getElementById('output');
    return outputEl;
  }

  /**
   * 打印一行文本到输出区
   * @param {string} text - 要打印的文本
   * @param {string} [type] - 类型: '' | 'error' | 'hint' | 'important' | 'warning' | 'system' | 'evidence' | 'digital-human' | 'zheng-qiao'
   * @param {object} [opts] - { instant: true 跳过打字机效果, raw: true 保留 HTML }
   */
  function print(text, type = '', opts = {}) {
    const el = ensureOutput();
    const line = document.createElement('div');
    line.className = `line ${type}`;
    if (opts.raw) {
      line.innerHTML = text;
    } else {
      line.textContent = text;
    }
    el.appendChild(line);
    el.scrollTop = el.scrollHeight;
    if (type === 'evidence') {
      line.classList.add('evidence-flash');
      setTimeout(() => line.classList.remove('evidence-flash'), 700);
    }
    return line;
  }

  /**
   * 打印多行（数组）
   */
  function printLines(lines, type = '') {
    lines.forEach(line => print(line, type));
  }

  /**
   * 清屏（保留 header）
   */
  function clear() {
    const el = ensureOutput();
    el.innerHTML = '';
  }

  /**
   * 屏幕闪烁（郑桥介入）
   */
  function shakeScreen() {
    const term = document.getElementById('terminal');
    if (!term) return;
    term.classList.add('shake');
    setTimeout(() => term.classList.remove('shake'), 700);
  }

  /**
   * 阶段过渡动画：黑屏 1s + 显示阶段标题
   */
  async function stageTransition(stageNum, stageName) {
    const el = ensureOutput();
    el.style.transition = 'opacity 0.5s';
    el.style.opacity = '0';
    await new Promise(r => setTimeout(r, 600));
    print(`━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━`, 'system');
    print(`[SYSTEM] 进入阶段 ${stageNum}：${stageName}`, 'system');
    print(`━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━`, 'system');
    el.style.opacity = '1';
    el.scrollTop = el.scrollHeight;
  }

  /**
   * 启动画面（淡入）
   */
  function startScreen() {
    print('', '');
    print('[终端初始化...]', 'hint');
    setTimeout(() => print('[连接成功]', 'hint'), 500);
  }

  /**
   * 输入框工具
   */
  function getInput() { return document.getElementById('command-input'); }
  function focusInput() { getInput()?.focus(); }
  function clearInput() { const el = getInput(); if (el) el.value = ''; }
  function getInputValue() { return getInput()?.value || ''; }

  return { print, printLines, clear, shakeScreen, stageTransition, startScreen, getInput, focusInput, clearInput, getInputValue };
})();
```

```javascript
// GAME 16 — 游戏引擎
// 职责：GameState 管理、阶段逻辑、Combine、Ending、持久化

const SAVE_KEY = 'game16-save-v1';
const VERSION = '1.0.0';

const game = (() => {
  // 初始状态
  function freshState() {
    return {
      currentStage: 1,
      stageIntroShown: { 1: false, 2: false, 3: false, 4: false, 5: false },
      unlockedEvidence: [],
      evidenceViewed: {},
      combineUnlocked: [],
      unlockedSystems: ['OA'],  // OA 初始可用
      gymNameEntered: false,
      phoneUnlocked: false,
      backupPrompted: false,
      backupCreated: false,
      endingReached: null,
      playerChoice: null,
      startTime: Date.now(),
      saveTime: 0,
      totalCommands: 0,
      version: VERSION,
    };
  }

  let state = freshState();

  function getState() { return state; }
  function resetState() { state = freshState(); return state; }

  // 持久化
  function save() {
    try {
      state.saveTime = Date.now();
      localStorage.setItem(SAVE_KEY, JSON.stringify({ state, version: VERSION }));
      return true;
    } catch (e) {
      console.error('Save failed:', e);
      return false;
    }
  }

  function load() {
    try {
      const raw = localStorage.getItem(SAVE_KEY);
      if (!raw) return null;
      const data = JSON.parse(raw);
      if (data.version !== VERSION) {
        console.warn('Save version mismatch, ignoring');
        return null;
      }

// GAME 16 — 命令分发
// 解析玩家输入、路由到对应处理函数

const command = (() => {
  // 命令注册表
  const commands = {};
  const aliases = {
    '?': 'help', 'h': 'help',
    'l': 'list', 'ls': 'list',
    'a': 'access', 'enter': 'access',
    'c': 'combine',
    'cls': 'clear',
    'conclusion': 'conclusions', 'summary': 'conclusions',
    'save': 'backup', 'b': 'backup',
  };

  // 注册一个命令
  function register(name, def) {
    commands[name] = def;
  }

  // 解析输入
  function parseInput(raw) {
    const trimmed = raw.trim();
    if (!trimmed) return null;
    const parts = trimmed.split(/\s+/);
    let cmd = parts[0].toLowerCase();
    if (aliases[cmd]) cmd = aliases[cmd];
    return { cmd, args: parts.slice(1), raw: trimmed };
  }

  // 派发
  function dispatch(input) {
    const parsed = parseInput(input);
    if (!parsed) return;

    const state = game.getState();
    state.totalCommands++;

    const def = commands[parsed.cmd];
    if (!def) {
      ui.print(`未知命令：${parsed.cmd}`, 'error');
      ui.print('输入 help 查看可用命令。', 'hint');
      return;
    }

    if (def.unlockedWhen && !def.unlockedWhen(state)) {
      ui.print('该命令暂未解锁。', 'error');
      return;
    }

    if (def.requiresArgs && parsed.args.length === 0) {
      ui.print(`命令 ${parsed.cmd} 需要参数。用法：${def.usage || parsed.cmd}`, 'error');
      return;
    }

    def.fn(parsed.args, parsed.raw);
    game.save();  // 每次命令后自动保存
  }

  return { register, parseInput, dispatch };
})();

  if (choice === 'accept') {

  } else if (choice === 'decline') {

    ui.print('郑桥：你确定？那就别怪我了。', 'zheng-qiao');
    setTimeout(() => {

        //   unlockedWhen: (state) => state.backupCreated && state.endingReached === 'ending2_trigger'
        // 这里只需直接进入 submit 处理
        game.getState()._waitingFor = 'submit-or-not';
      } else {
        // 结局 4
        game.getState().endingReached = 'ending4';
        showEnding('ending4');

    }, 1500);

      game.getState().endingReached = 'ending3';
